What are Events?

Events are things that happen in your webpage - like clicks, key presses, or page loading. JavaScript can "listen" for these events and respond!

Common Event Types

Here are the most useful events you'll work with:

Mouse Events

  • click - Mouse click
  • dblclick - Double click
  • mouseover - Mouse enters
  • mouseout - Mouse leaves
  • mousemove - Mouse moves

Keyboard Events

  • keydown - Key pressed down
  • keyup - Key released
  • keypress - Key pressed

Form Events

  • submit - Form submitted
  • change - Value changed
  • focus - Element focused
  • blur - Lost focus

Event Listeners

There are three ways to handle events:

// Method 1: Inline HTML (not recommended)
<button onclick="alert('Clicked!')">Click Me</button>

// Method 2: DOM property
button.onclick = function() {
    alert('Clicked!');
};

// Method 3: addEventListener (BEST WAY!)
button.addEventListener('click', function() {
    alert('Clicked!');
});
Why use addEventListener?
  • Can add multiple handlers to same event
  • Can remove handlers later
  • More control over event behavior
  • Works with all event types

Multiple Event Listeners:

let button = document.getElementById('multi-button');

button.addEventListener('click', function() {
    console.log('First handler');
});

button.addEventListener('click', function() {
    console.log('Second handler');
});

// Both will run!

The Event Object

When an event happens, JavaScript creates an event object with useful information:

element.addEventListener('click', function(event) {
    console.log(event.type);        // 'click'
    console.log(event.target);      // The clicked element
    console.log(event.clientX);     // Mouse X position
    console.log(event.clientY);     // Mouse Y position
    
    event.preventDefault();         // Stop default behavior
    event.stopPropagation();       // Stop event bubbling
});

The Complete Game

Create a new file called whack-a-mole.html:

<!DOCTYPE html>
<html>
<head>
    <title>Whack-a-Mole Game</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            background: linear-gradient(135deg, #667eea 0%, #764ba2 100%);
            margin: 0;
            padding: 20px;
        }
        
        .game-container {
            max-width: 600px;
            margin: 0 auto;
            background: white;
            padding: 30px;
            border-radius: 20px;
            box-shadow: 0 10px 30px rgba(0,0,0,0.3);
        }
        
        .game-board {
            display: grid;
            grid-template-columns: repeat(3, 1fr);
            gap: 15px;
            margin: 20px 0;
        }
        
        .hole {
            width: 120px;
            height: 120px;
            background: #8B4513;
            border-radius: 50%;
            position: relative;
            overflow: hidden;
            cursor: pointer;
            box-shadow: inset 0 10px 20px rgba(0,0,0,0.5);
        }
        
        .mole {
            width: 80px;
            height: 80px;
            background: #333;
            border-radius: 50%;
            position: absolute;
            bottom: -80px;
            left: 50%;
            transform: translateX(-50%);
            transition: bottom 0.3s;
        }
        
        .mole::before {
            content: 'üëø';
            font-size: 60px;
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
        }
        
        .mole.up {
            bottom: 20px;
        }
        
        .score-board {
            display: flex;
            justify-content: space-around;
            margin: 20px 0;
            font-size: 20px;
        }
        
        button {
            font-size: 20px;
            padding: 10px 30px;
            background-color: #667eea;
            color: white;
            border: none;
            border-radius: 10px;
            cursor: pointer;
            transition: background-color 0.3s;
        }
        
        button:hover {
            background-color: #764ba2;
        }
        
        button:disabled {
            background-color: #999;
            cursor: not-allowed;
        }
        
        #timer {
            font-size: 24px;
            color: #e74c3c;
            font-weight: bold;
        }
    </style>
</head>
<body>
    <div class="game-container">
        <h1>üî® Whack-a-Mole! üî®</h1>
        
        <div class="score-board">
            <div>Score: <span id="score">0</span></div>
            <div id="timer">Time: 30s</div>
            <div>High Score: <span id="high-score">0</span></div>
        </div>
        
        <div class="game-board" id="game-board">
            <!-- Holes will be created here -->
        </div>
        
        <button id="start-btn" onclick="startGame()">Start Game</button>
    </div>

    <script>
        let score = 0;
        let timeLeft = 30;
        let gameActive = false;
        let moleTimer;
        let countdownTimer;
        let highScore = localStorage.getItem('whackHighScore') || 0;
        
        // Display high score
        document.getElementById('high-score').textContent = highScore;
        
        // Create the game board
        function createBoard() {
            const board = document.getElementById('game-board');
            board.innerHTML = '';
            
            for (let i = 0; i < 9; i++) {
                const hole = document.createElement('div');
                hole.className = 'hole';
                hole.id = `hole-${i}`;
                
                const mole = document.createElement('div');
                mole.className = 'mole';
                
                // Add click event to mole
                mole.addEventListener('click', whackMole);
                
                hole.appendChild(mole);
                board.appendChild(hole);
            }
        }
        
        // Show random mole
        function showRandomMole() {
            if (!gameActive) return;
            
            // Hide all moles
            document.querySelectorAll('.mole').forEach(mole => {
                mole.classList.remove('up');
            });
            
            // Pick random hole
            const randomHole = Math.floor(Math.random() * 9);
            const mole = document.querySelector(`#hole-${randomHole} .mole`);
            mole.classList.add('up');
            
            // Hide after random time
            const hideTime = Math.random() * 1000 + 500; // 0.5-1.5 seconds
            setTimeout(() => {
                mole.classList.remove('up');
            }, hideTime);
        }
        
        // Whack the mole
        function whackMole(event) {
            if (!gameActive) return;
            if (!event.target.classList.contains('up')) return;
            
            score++;
            document.getElementById('score').textContent = score;
            event.target.classList.remove('up');
            
            // Visual feedback
            event.target.style.background = '#ff6b6b';
            setTimeout(() => {
                event.target.style.background = '#333';
            }, 200);
        }
        
        // Start the game
        function startGame() {
            score = 0;
            timeLeft = 30;
            gameActive = true;
            
            document.getElementById('score').textContent = score;
            document.getElementById('start-btn').disabled = true;
            
            // Start spawning moles
            moleTimer = setInterval(showRandomMole, 1000);
            
            // Start countdown
            countdownTimer = setInterval(() => {
                timeLeft--;
                document.getElementById('timer').textContent = `Time: ${timeLeft}s`;
                
                if (timeLeft <= 0) {
                    endGame();
                }
            }, 1000);
        }
        
        // End the game
        function endGame() {
            gameActive = false;
            clearInterval(moleTimer);
            clearInterval(countdownTimer);
            
            document.getElementById('start-btn').disabled = false;
            
            // Hide all moles
            document.querySelectorAll('.mole').forEach(mole => {
                mole.classList.remove('up');
            });
            
            // Check high score
            if (score > highScore) {
                highScore = score;
                localStorage.setItem('whackHighScore', highScore);
                document.getElementById('high-score').textContent = highScore;
                alert(`New High Score: ${score}! üéâ`);
            } else {
                alert(`Game Over! Your score: ${score}`);
            }
        }
        
        // Initialize game
        createBoard();
        
        // Keyboard shortcut
        document.addEventListener('keydown', function(event) {
            if (event.key === ' ' && !gameActive) {
                event.preventDefault();
                startGame();
            }
        });
    </script>
</body>
</html>

Event Handling in the Game:

  • Click events on moles to "whack" them
  • Button click to start the game
  • Keyboard event (spacebar) as shortcut
  • Timer events with setInterval
  • Event delegation for dynamically created elements
