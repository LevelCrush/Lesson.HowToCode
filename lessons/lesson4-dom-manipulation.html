What is the DOM?

DOM stands for Document Object Model. It's how JavaScript sees and interacts with your HTML.

Think of it like this:

  • HTML creates the elements
  • CSS styles them
  • JavaScript can find, change, add, or remove them!
Click the button to see DOM manipulation!

Selecting Elements

Before you can change something, you need to find it:

// Select by ID (most specific)
let title = document.getElementById('myTitle');

// Select by class (returns all matching elements)
let buttons = document.getElementsByClassName('btn');

// Select by tag name
let paragraphs = document.getElementsByTagName('p');

// Modern way: querySelector (uses CSS selectors)
let firstButton = document.querySelector('.btn');
let allButtons = document.querySelectorAll('.btn');
Pro Tip: querySelector is the most flexible:
  • querySelector('#myId') - select by ID
  • querySelector('.myClass') - select by class
  • querySelector('button') - select by tag
  • querySelector('#container .btn') - complex selectors!

Changing Elements

Once you've selected an element, you can change it:

Change Text Content

let heading = document.getElementById('title');
heading.textContent = "New Title!";

// HTML content (can include tags)
heading.innerHTML = "New Title!";

Change Styles

let box = document.getElementById('myBox');
box.style.backgroundColor = 'red';
box.style.width = '200px';
box.style.fontSize = '20px';

Creating & Removing Elements

You can dynamically add or remove elements:

// Create a new element
let newDiv = document.createElement('div');
newDiv.textContent = "I'm new!";
newDiv.style.color = 'blue';

// Add it to the page
document.body.appendChild(newDiv);

// Remove an element
let oldElement = document.getElementById('remove-me');
oldElement.remove();

The Complete Game

Create a new file called color-game.html:

<!DOCTYPE html>
<html>
<head>
    <title>Click the Color Game</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
            background-color: #f0f0f0;
        }
        
        .game-container {
            max-width: 600px;
            margin: 50px auto;
            background: white;
            padding: 30px;
            border-radius: 15px;
            box-shadow: 0 5px 15px rgba(0,0,0,0.1);
        }
        
        #color-display {
            font-size: 48px;
            font-weight: bold;
            margin: 20px;
            padding: 20px;
            background-color: #f8f9fa;
            border-radius: 10px;
        }
        
        .color-grid {
            display: grid;
            grid-template-columns: repeat(3, 1fr);
            gap: 15px;
            margin: 20px 0;
        }
        
        .color-box {
            height: 100px;
            border-radius: 10px;
            cursor: pointer;
            transition: transform 0.2s;
        }
        
        .color-box:hover {
            transform: scale(1.05);
        }
        
        #score-display {
            font-size: 24px;
            margin: 20px;
        }
        
        #message {
            font-size: 20px;
            height: 30px;
            margin: 10px;
        }
        
        .correct { color: #28a745; }
        .wrong { color: #dc3545; }
        
        button {
            font-size: 18px;
            padding: 10px 30px;
            background-color: #007bff;
            color: white;
            border: none;
            border-radius: 5px;
            cursor: pointer;
        }
        
        button:hover {
            background-color: #0056b3;
        }
    </style>
</head>
<body>
    <div class="game-container">
        <h1>ðŸŽ¨ Click the Color! ðŸŽ¨</h1>
        <p>Click the box that matches this color:</p>
        
        <div id="color-display">RGB(255, 0, 0)</div>
        
        <div class="color-grid" id="color-grid">
            <!-- Color boxes will be created here -->
        </div>
        
        <div id="message"></div>
        <div id="score-display">Score: <span id="score">0</span></div>
        
        <button onclick="newRound()">New Colors</button>
    </div>

    <script>
        let score = 0;
        let correctColor;
        
        // Generate random RGB color
        function randomColor() {
            let r = Math.floor(Math.random() * 256);
            let g = Math.floor(Math.random() * 256);
            let b = Math.floor(Math.random() * 256);
            return `RGB(${r}, ${g}, ${b})`;
        }
        
        // Create color boxes
        function createColorBoxes() {
            let grid = document.getElementById('color-grid');
            grid.innerHTML = ''; // Clear existing boxes
            
            // Pick a random position for the correct answer
            let correctPosition = Math.floor(Math.random() * 6);
            
            for (let i = 0; i < 6; i++) {
                let box = document.createElement('div');
                box.className = 'color-box';
                
                // Set color
                let color;
                if (i === correctPosition) {
                    color = correctColor;
                } else {
                    color = randomColor();
                }
                
                box.style.backgroundColor = color;
                
                // Add click event
                box.onclick = function() {
                    checkAnswer(color);
                };
                
                grid.appendChild(box);
            }
        }
        
        // Check if clicked color is correct
        function checkAnswer(clickedColor) {
            let message = document.getElementById('message');
            
            if (clickedColor.toUpperCase() === correctColor.toUpperCase()) {
                message.textContent = "Correct! Well done!";
                message.className = 'correct';
                score++;
                document.getElementById('score').textContent = score;
                
                // Auto new round after 1 second
                setTimeout(newRound, 1000);
            } else {
                message.textContent = "Wrong! Try again!";
                message.className = 'wrong';
            }
        }
        
        // Start a new round
        function newRound() {
            correctColor = randomColor();
            document.getElementById('color-display').textContent = correctColor;
            document.getElementById('message').textContent = '';
            createColorBoxes();
        }
        
        // Start the game
        newRound();
    </script>
</body>
</html>

How the game works:

  1. Generate a random RGB color as the target
  2. Create 6 color boxes using createElement
  3. One box gets the correct color, others get random colors
  4. Use onclick to check if player clicked the right one
  5. Update score and message using DOM manipulation
