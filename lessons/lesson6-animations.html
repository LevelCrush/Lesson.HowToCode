CSS Animations

CSS animations let you animate elements without JavaScript!

@keyframes bounce {
    0% { transform: translateY(0px); }
    50% { transform: translateY(-30px); }
    100% { transform: translateY(0px); }
}

.bouncing {
    animation: bounce 1s infinite;
}

See it in action:

Animation Properties:
  • animation-name: The @keyframes name
  • animation-duration: How long it takes
  • animation-timing-function: Speed curve (ease, linear)
  • animation-iteration-count: How many times (infinite)
  • animation-direction: Forward, reverse, alternate

CSS Transitions

Transitions create smooth changes between two states:

.box {
    width: 100px;
    height: 100px;
    background: blue;
    transition: all 0.3s ease;
}

.box:hover {
    width: 150px;
    background: red;
    transform: rotate(45deg);
}

JavaScript Animations

For complex animations, JavaScript gives you full control:

// Simple animation with setInterval
let position = 0;
let element = document.getElementById('moving-box');

setInterval(function() {
    position += 2;
    element.style.left = position + 'px';
    
    // Reset when off screen
    if (position > window.innerWidth) {
        position = -100;
    }
}, 20);

requestAnimationFrame

The best way to create smooth animations in JavaScript:

function animate() {
    // Update animation
    position += speed;
    element.style.left = position + 'px';
    
    // Continue animation
    requestAnimationFrame(animate);
}

// Start animation
requestAnimationFrame(animate);
Why use requestAnimationFrame?
  • Syncs with browser refresh rate (usually 60 FPS)
  • Pauses when tab is not visible (saves battery)
  • Smoother than setInterval
  • Better performance

The Complete Game

Create a new file called bouncing-ball-game.html:

<!DOCTYPE html>
<html>
<head>
    <title>Catch the Bouncing Ball</title>
    <style>
        body {
            margin: 0;
            padding: 0;
            font-family: Arial, sans-serif;
            overflow: hidden;
            background: linear-gradient(to bottom, #87CEEB, #98D8C8);
        }
        
        #game-area {
            position: relative;
            width: 100vw;
            height: 100vh;
            cursor: none;
        }
        
        #ball {
            position: absolute;
            width: 50px;
            height: 50px;
            background: radial-gradient(circle at 30% 30%, #ff6b6b, #c44569);
            border-radius: 50%;
            box-shadow: 0 5px 15px rgba(0,0,0,0.3);
            transition: transform 0.1s;
        }
        
        #ball.caught {
            transform: scale(0);
        }
        
        #paddle {
            position: absolute;
            width: 120px;
            height: 20px;
            background: #2c3e50;
            border-radius: 10px;
            bottom: 50px;
            left: 50%;
            transform: translateX(-50%);
            box-shadow: 0 5px 10px rgba(0,0,0,0.3);
        }
        
        #score-board {
            position: absolute;
            top: 20px;
            left: 20px;
            font-size: 24px;
            color: white;
            text-shadow: 2px 2px 4px rgba(0,0,0,0.5);
        }
        
        #game-over {
            position: absolute;
            top: 50%;
            left: 50%;
            transform: translate(-50%, -50%);
            background: white;
            padding: 30px;
            border-radius: 20px;
            text-align: center;
            display: none;
            box-shadow: 0 10px 30px rgba(0,0,0,0.3);
        }
        
        button {
            font-size: 20px;
            padding: 10px 30px;
            background: #3498db;
            color: white;
            border: none;
            border-radius: 10px;
            cursor: pointer;
            margin-top: 20px;
        }
        
        button:hover {
            background: #2980b9;
        }
        
        .particle {
            position: absolute;
            width: 10px;
            height: 10px;
            background: #f39c12;
            border-radius: 50%;
            pointer-events: none;
        }
        
        @keyframes explode {
            to {
                transform: translate(var(--x), var(--y)) scale(0);
                opacity: 0;
            }
        }
    </style>
</head>
<body>
    <div id="game-area">
        <div id="ball"></div>
        <div id="paddle"></div>
        <div id="score-board">
            Score: <span id="score">0</span> | Misses: <span id="misses">0</span>
        </div>
        <div id="game-over">
            <h2>Game Over!</h2>
            <p>Final Score: <span id="final-score">0</span></p>
            <button onclick="resetGame()">Play Again</button>
        </div>
    </div>

    <script>
        // Game variables
        let ball = document.getElementById('ball');
        let paddle = document.getElementById('paddle');
        let gameArea = document.getElementById('game-area');
        let score = 0;
        let misses = 0;
        let ballX = 100;
        let ballY = 100;
        let velocityX = 5;
        let velocityY = 0;
        let gravity = 0.5;
        let bounceSpeed = -15;
        let gameRunning = true;

        // Update paddle position with mouse
        gameArea.addEventListener('mousemove', function(e) {
            if (!gameRunning) return;
            let paddleX = e.clientX - 60; // Center paddle on cursor
            paddle.style.left = paddleX + 'px';
        });

        // Create particle effect
        function createParticles(x, y) {
            for (let i = 0; i < 10; i++) {
                let particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = x + 'px';
                particle.style.top = y + 'px';
                particle.style.setProperty('--x', (Math.random() - 0.5) * 100 + 'px');
                particle.style.setProperty('--y', (Math.random() - 0.5) * 100 + 'px');
                particle.style.animation = 'explode 0.5s ease-out forwards';
                gameArea.appendChild(particle);
                
                setTimeout(() => particle.remove(), 500);
            }
        }

        // Game loop
        function gameLoop() {
            if (!gameRunning) return;

            // Update ball position
            ballX += velocityX;
            ballY += velocityY;
            velocityY += gravity;

            // Bounce off walls
            if (ballX <= 0 || ballX >= window.innerWidth - 50) {
                velocityX = -velocityX;
            }

            // Check paddle collision
            let paddleRect = paddle.getBoundingClientRect();
            let ballRect = ball.getBoundingClientRect();

            if (ballRect.bottom >= paddleRect.top && 
                ballRect.bottom <= paddleRect.bottom &&
                ballRect.right >= paddleRect.left && 
                ballRect.left <= paddleRect.right &&
                velocityY > 0) {
                
                // Ball caught!
                velocityY = bounceSpeed;
                score++;
                document.getElementById('score').textContent = score;
                
                // Speed up game
                velocityX *= 1.05;
                
                // Particle effect
                createParticles(ballX + 25, ballY + 25);
            }

            // Check if ball fell
            if (ballY > window.innerHeight) {
                misses++;
                document.getElementById('misses').textContent = misses;
                
                if (misses >= 3) {
                    endGame();
                } else {
                    // Reset ball
                    ballX = Math.random() * (window.innerWidth - 50);
                    ballY = 100;
                    velocityY = 0;
                }
            }

            // Update ball position
            ball.style.left = ballX + 'px';
            ball.style.top = ballY + 'px';

            requestAnimationFrame(gameLoop);
        }

        // End game
        function endGame() {
            gameRunning = false;
            document.getElementById('final-score').textContent = score;
            document.getElementById('game-over').style.display = 'block';
            gameArea.style.cursor = 'auto';
        }

        // Reset game
        function resetGame() {
            score = 0;
            misses = 0;
            ballX = 100;
            ballY = 100;
            velocityX = 5;
            velocityY = 0;
            gameRunning = true;
            
            document.getElementById('score').textContent = score;
            document.getElementById('misses').textContent = misses;
            document.getElementById('game-over').style.display = 'none';
            gameArea.style.cursor = 'none';
            
            gameLoop();
        }

        // Start game
        gameLoop();
    </script>
</body>
</html>

  • requestAnimationFrame for smooth 60 FPS gameplay
  • Physics simulation with gravity and velocity
  • CSS transitions for ball scaling effect
  • Particle animations with CSS keyframes
  • Dynamic positioning following mouse cursor
